27 void ITHACAPOD::getModes(PtrList<volVectorField>& snapshotsU, PtrList<volVectorField>& modes, bool podex, bool supex, bool sup, int nmodes)
28 {
29 
30  if (nmodes == 0)
31  {
32  nmodes = snapshotsU.size();
33  }
34 
35  if (podex == 0)
36  {
37  PtrList<volVectorField> Bases;
38  Bases.resize(nmodes);
39  modes.resize(nmodes);
40  Eigen::MatrixXd _corMatrix;
41  scalarField eigenValues(snapshotsU.size());
42  scalarField cumEigenValues(snapshotsU.size());
43  List<scalarField> eigenVector(snapshotsU.size());
44 
45  for (label i = 0; i < snapshotsU.size(); i++)
46  {
47  eigenVector[i].setSize(snapshotsU.size());
48  }
49  _corMatrix = ITHACAPOD::corMatrix(snapshotsU);
50  Info << "####### Performing the POD decomposition for " << snapshotsU[0].name() << " #######" << endl;
51  Eigen::EigenSolver<Eigen::MatrixXd> es(_corMatrix);
52  Info << "####### End of the POD decomposition for " << snapshotsU[0].name() << " #######" << endl;
53  Eigen::MatrixXd eigenVectoreig = es.eigenvectors().real();
54 
55  cumEigenValues[0] = es.eigenvalues().real()(0) / es.eigenvalues().real().sum();
56  eigenValues[0] = es.eigenvalues().real()(0) / es.eigenvalues().real().sum();
57  for (label i = 1; i < snapshotsU.size(); i++)
58  {
59  cumEigenValues[i] = cumEigenValues[i - 1] + es.eigenvalues().real()(i) / es.eigenvalues().real().sum();
60  eigenValues[i] = es.eigenvalues().real()(i) / es.eigenvalues().real().sum();
61  }
62 
63  for (label i = 0; i < snapshotsU.size(); i++)
64  {
65  for (label k = 0; k < snapshotsU.size(); k++)
66  {
67  eigenVector[i][k] = eigenVectoreig(k, i);
68  }
69  }
70 
71  volVectorField tmb_bu
72  (
73  snapshotsU[0].name(),
74  snapshotsU[0]
75  );
76  for (label i = 0; i < nmodes; i++)
77  {
78  volVectorField tmb_bu(eigenVector[i][0]*snapshotsU[0]);
79  for (label k = 0; k < snapshotsU.size(); k++)
80  {
81  tmb_bu += eigenVector[i][k] * snapshotsU[k];
82  }
83  Bases.set(i, tmb_bu);
84  Info << "creating the bases " << i << " for " << snapshotsU[0].name() << endl;
85  }
86  ITHACAPOD::normalizeBases(Bases);
87  for (label i = 0; i < Bases.size(); i++)
88  {
89  volVectorField tmp
90  (
91  snapshotsU[0].name(),
92  Bases[i]
93  );
94  modes.set(i, tmp);
95  }
96  Info << "####### Saving the POD bases for " << snapshotsU[0].name() << " #######" << endl;
97  ITHACAPOD::exportBases(modes, snapshotsU, sup);
98  ITHACAPOD::exportEigenvalues(eigenValues, snapshotsU[0].name());
99  ITHACAPOD::exportcumEigenvalues(cumEigenValues, snapshotsU[0].name());
100  }
101  else
102  {
103  Info << "Reading the existing modes" << endl;
104  if (sup == 1)
105  {
106  ITHACAstream::read_fields(modes, snapshotsU[0], "./ITHACAoutput/supremizer/");
107  }
108  else
109  {
110  ITHACAstream::read_fields (modes, snapshotsU[0], "./ITHACAoutput/POD/");
111  }
112  }
113 }

201 void ITHACAPOD::normalizeBases(PtrList<volScalarField>& Bases)
202 {
203  scalar magSumSquare;
204  for (label j = 0; j < Bases.size(); j++)
205  {
206  magSumSquare = Foam::sqrt(fvc::domainIntegrate(Bases[j] * Bases[j]).value());
207  if (magSumSquare > SMALL)
208  {
209  Bases[j] /= magSumSquare;
210  }
211  Bases[j].correctBoundaryConditions();
212  }
213 }

246 Eigen::MatrixXd ITHACAPOD::corMatrix(PtrList<volScalarField>& snapshots)
247 {
248  Info << "########## Filling the correlation matrix for " << snapshots[0].name() << "##########" << endl;
249  Eigen::MatrixXd matrix( snapshots.size(), snapshots.size());
250  for (label i = 0; i < snapshots.size(); i++)
251  {
252  Info << "Filling row " << i << " of the " << snapshots[0].name() << " correlation matrix" << endl;
253  for (label j = 0; j <= i; j++)
254  {
255  matrix(i, j) = fvc::domainIntegrate(snapshots[i] * snapshots[j]).value();
256  }
257  }
258  for (label i = 1; i < snapshots.size(); i++)
259  {
260  for (label j = 0; j < i; j++)
261  {
262  matrix(j, i) = matrix(i, j);
263  }
264  }
265  return matrix;
266 }
