162 Eigen::MatrixXd ITHACAutilities::get_mass_matrix(PtrList<volVectorField>& modes)
163 {
164  label Msize = modes.size();
165  Eigen::MatrixXd M_matrix(Msize, Msize);
166 
167  // Project everything
168  for (label i = 0; i < Msize; i++)
169  {
170  for (label j = 0; j < Msize; j++)
171  {
172  M_matrix(i, j) = fvc::domainIntegrate(modes[i] & modes[j]).value();
173  }
174  }

194 Eigen::VectorXd ITHACAutilities::get_coeffs(volVectorField snapshot, PtrList<volVectorField>& modes)
195 {
196  label Msize = modes.size();
197  Eigen::MatrixXd M_matrix = get_mass_matrix(modes);
198  Eigen::VectorXd a(Msize);
199 
200  Eigen::VectorXd b(Msize);
201 
202  // Project everything

262 Eigen::MatrixXd ITHACAutilities::foam2eigen(PtrList<volVectorField>& fields1)
263 {
264  Eigen::MatrixXd out;
265  out.resize(int(fields1[0].size()*3),fields1.size());
266 
267  for(int k=0; k<fields1.size(); k++)
268  {
269  for(int l=0; l<fields1[k].size(); l++)
270  {
271  out(3*l,k) = fields1[k][l][0];
272  out(3*l+1,k) = fields1[k][l][1];
273  out(3*l+2,k) = fields1[k][l][2];
274  }
275  }

294 void ITHACAutilities::setBoxToValue(volScalarField& field, Eigen::MatrixXd Box, double value)
295 {
296  for (label i = 0; i < field.internalField().size(); i++)
297  {
298  auto cx = field.mesh().C()[i].component(vector::X);
299  auto cy = field.mesh().C()[i].component(vector::Y);
300  auto cz = field.mesh().C()[i].component(vector::Z);
301  if (cx >= Box(0, 0) && cy >= Box(0, 1) && cz >= Box(0, 2) && cx <= Box(1, 0) && cy <= Box(1, 1) && cz <= Box(1, 2) )
302  {
303  field.ref()[i] = value;
304  }
305  }
306  for (label i = 0; i < field.boundaryField().size(); i++)
307  {
308  for (label j = 0; j < field.boundaryField()[i].size(); j++)
309  {
310  if (field.boundaryField()[i].type() == "fixedValue" || field.boundaryField()[i].type() == "calculated")
311  {
312  auto cx = field.mesh().C().boundaryField()[i][j][0];
313  auto cy = field.mesh().C().boundaryField()[i][j][1];
314  auto cz = field.mesh().C().boundaryField()[i][j][2];
315  if (cx >= Box(0, 0) && cy >= Box(0, 1) && cz >= Box(0, 2) && cx <= Box(1, 0) && cy <= Box(1, 1) && cz <= Box(1, 2) )
316  {
317  field.boundaryFieldRef()[i][j] = value;
318  }
319  }
320  }
321  }
322 }
